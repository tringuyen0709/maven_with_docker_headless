pipeline {
    agent any
    
    options {
        disableConcurrentBuilds(abortPrevious: true)
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build Docker Image') {
            steps {
                // Build image từ Dockerfile trong thư mục hiện tại
                sh 'docker build -t selenium-docker-test .'
            }
        }

        stage('Run Selenium Tests') {
            steps {
                /* 
                   LƯU Ý: Phải dùng -v (volume) để map thư mục target từ container ra ngoài Jenkins.
                   Nếu không có -v, bước 'junit' ở dưới sẽ không tìm thấy file kết quả vì nó nằm trong container đã bị xóa.
                */
                sh 'docker run --rm -v ${WORKSPACE}/target:/app/target selenium-docker-test mvn test'
            }
        }
    }

    post {
        always {
            echo 'Pipeline finished'
            // Thu thập kết quả test từ thư mục target đã được mount ra ngoài
            junit '**/target/surefire-reports/*.xml'
        }
        success {
            echo 'All tests passed'
        }
        failure {
            echo 'Some tests failed'
        }
    }
}
